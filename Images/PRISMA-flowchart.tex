    start chain=going below,
 mynode/.style = {
        draw, rectangle, align=center, text width=4cm,
        font=\small, inner sep=3ex, outer sep=0pt,
        on chain},
mylabel/.style = {
        draw, rectangle, align=center, rounded corners, 
        font=\small\bfseries, inner sep=1ex, outer sep=0pt,
        fill=cyan!30, minimum height=32mm,
        on chain},

     arrow/.style = {very thick,-stealth}
                    ]

\node (n3)  [mynode,join]   {289 titles screened};
\node (n4)  [mynode,join]   {217 abstracts screened};

\draw[arrow] (n3) -- (n3r);
